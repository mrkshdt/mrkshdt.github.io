def monte(ticker, t_intervals=100, iterations=25):
    # get stock data
    data = pd.DataFrame()
    data[ticker] = wb.DataReader(ticker, data_source='yahoo', start='2020-07-01')['Adj Close']

    # definition and calculation of daily returns
    log_returns = np.log(1 + data.pct_change())
    u = log_returns.mean()
    var = log_returns.var()
    drift = u - (0.1 * var)
    stdev = log_returns.std()
    drift.values
    stdev.values
    daily_returns = np.exp(drift.values + stdev.values * norm.ppf(np.random.rand(t_intervals, iterations),loc=-0.1))
    
    # Set start and simulate closing prices 
    S0 = data.iloc[-1]
    price_list = np.zeros_like(daily_returns)
    price_list[0] = S0
    for t in range(1, t_intervals):
        price_list[t] = price_list[t - 1] * daily_returns[t]
        
    # bring list into correct format
    def list_correct(data,price_list):
        new = []
        for i in range(0,len(data)+len(price_list)):
            if i<len(data):
                tmp = [data.values[i] for z in range(iterations)]
                new.append(tmp)
            else:
                new.append(price_list[i-len(data)])       
        return new
                
    result = list_correct(data, price_list)
    
    # plot historical data and simulation
    def monte_carlo_plot(data,ticker):
        fig, ax = plt.subplots(figsize=(16,8))
        
        plt.title('Monte Carlo Simulation of $%s'%ticker, fontsize=24)
        plt.xlabel('Days traded',fontsize=18)
        plt.ylabel('Closing Price in USD ($)', fontsize=18)
    
        tmp = []
        for i in result:
            tmp.append(max(i))
    
        # stock data background
        ax.add_patch(Rectangle((len(data)-len(price_list),0), 100, 1/5*max(tmp)+max(tmp),alpha=0.2))
        ax.add_patch(Rectangle((0,0), len(data), 1/5*max(tmp)+max(tmp),alpha=0.2, facecolor='gray'))
        
        # text 'Future' and 'Historical'
        plt.text((0.75*len(tmp)), max(tmp), 'Future', fontsize=18)
        plt.text(0.2*len(result), max(tmp), 'Historical', fontsize=18)
    
        fig.set_facecolor('silver')
        
        ax.plot(data, linewidth=2)
        
    return monte_carlo_plot(result,ticker)
